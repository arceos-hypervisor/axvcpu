9/// Reasons for VM-Exits returned by [AxArchVCpu::run].
10///
11/// When a guest virtual CPU executes, various conditions can cause control to be
12/// transferred back to the hypervisor. This enum represents all possible exit reasons
13/// that can occur during VCpu execution.
14///
15/// # VM Exit Categories
16///
17/// - **I/O Operations**: MMIO reads/writes, port I/O, system register access
18/// - **System Events**: Hypercalls, interrupts, nested page faults
19/// - **Power Management**: CPU power state changes, system shutdown
20/// - **Multiprocessing**: IPI sending, secondary CPU bring-up
21/// - **Error Conditions**: Entry failures, invalid states
22///
23/// # Compatibility Note
24///
25/// This enum draws inspiration from [kvm-ioctls](https://github.com/rust-vmm/kvm-ioctls/blob/main/src/ioctls/vcpu.rs)
26/// for consistency with existing virtualization frameworks.
27#[non_exhaustive]
28#[derive(Debug)]
29pub enum AxVCpuExitReason {
30    /// A guest instruction triggered a hypercall to the hypervisor.
31    ///
32    /// Hypercalls are a mechanism for the guest OS to request services from
33    /// the hypervisor, similar to system calls in a traditional OS.
34    Hypercall {
35        /// The hypercall number identifying the requested service
36        nr: u64,
37        /// Arguments passed to the hypercall (up to 6 parameters)
38        args: [u64; 6],
39    },
40
41    /// The guest performed a Memory-Mapped I/O (MMIO) read operation.
42    ///
43    /// MMIO reads occur when the guest accesses device registers or other
44    /// hardware-mapped memory regions that require hypervisor emulation.
45    MmioRead {
46        /// Guest physical address being read from
47        addr: GuestPhysAddr,
48        /// Width/size of the memory access (8, 16, 32, or 64 bits)
49        width: AccessWidth,
50        /// Index of the guest register that will receive the read value
51        reg: usize,
52        /// Width of the destination register  
53        reg_width: AccessWidth,
54        /// Whether to sign-extend the read value to fill the register
55        signed_ext: bool,
56    },
57
58    /// The guest performed a Memory-Mapped I/O (MMIO) write operation.
59    ///
60    /// MMIO writes occur when the guest writes to device registers or other
61    /// hardware-mapped memory regions that require hypervisor emulation.
62    MmioWrite {
63        /// Guest physical address being written to
64        addr: GuestPhysAddr,
65        /// Width/size of the memory access (8, 16, 32, or 64 bits)
66        width: AccessWidth,
67        /// Data being written to the memory location
68        data: u64,
69    },
70
71    /// The guest performed a system register read operation.
72    ///
73    /// System registers are architecture-specific control and status registers:
74    /// - **x86_64**: Model-Specific Registers (MSRs)
75    /// - **RISC-V**: Control and Status Registers (CSRs)
76    /// - **AArch64**: System registers accessible via MRS instruction
77    SysRegRead {
78        /// Address/identifier of the system register being read
79        ///
80        /// - **x86_64/RISC-V**: Direct register address
81        /// - **AArch64**: ESR_EL2.ISS format (`<op0><op2><op1><CRn>00000<CRm>0`)
82        ///   compatible with the `aarch64_sysreg` crate numbering scheme
83        addr: SysRegAddr,
84        /// Index of the guest register that will receive the read value
85        ///
86        /// **Note**: Unused on x86_64 where the result is always stored in `[edx:eax]`
87        reg: usize,
88    },
89
90    /// The guest performed a system register write operation.
91    ///
92    /// System registers are architecture-specific control and status registers:
93    /// - **x86_64**: Model-Specific Registers (MSRs)
94    /// - **RISC-V**: Control and Status Registers (CSRs)
95    /// - **AArch64**: System registers accessible via MSR instruction  
96    SysRegWrite {
97        /// Address/identifier of the system register being written
98        ///
99        /// - **x86_64/RISC-V**: Direct register address
100        /// - **AArch64**: ESR_EL2.ISS format (`<op0><op2><op1><CRn>00000<CRm>0`)
101        ///   compatible with the `aarch64_sysreg` crate numbering scheme
102        addr: SysRegAddr,
103        /// Data being written to the system register
104        value: u64,
105    },
106
107    /// The guest performed a port-based I/O read operation.
108    ///
109    /// **Architecture**: x86-specific (other architectures don't have port I/O)
110    ///
111    /// The destination register is implicitly `al`/`ax`/`eax` based on the access width.
112    IoRead {
113        /// I/O port number being read from
114        port: Port,
115        /// Width of the I/O access (8, 16, or 32 bits)
116        width: AccessWidth,
117    },
118
119    /// The guest performed a port-based I/O write operation.
120    ///
121    /// **Architecture**: x86-specific (other architectures don't have port I/O)
122    ///
123    /// The source register is implicitly `al`/`ax`/`eax` based on the access width.
124    IoWrite {
125        /// I/O port number being written to
126        port: Port,
127        /// Width of the I/O access (8, 16, or 32 bits)
128        width: AccessWidth,
129        /// Data being written to the I/O port
130        data: u64,
131    },
132
133    /// An external interrupt was delivered to the VCpu.
134    ///
135    /// This represents hardware interrupts from external devices that need
136    /// to be processed by the guest or hypervisor.
137    ///
138    /// **Note**: This enum may be extended with additional fields in the future.
139    /// Use `..` in pattern matching to ensure forward compatibility.
140    ExternalInterrupt {
141        /// Hardware interrupt vector number
142        vector: u64,
143    },
144
145    /// A nested page fault occurred during guest memory access.
146    ///
147    /// Also known as EPT violations on x86. These faults occur when:
148    /// - Guest accesses unmapped memory regions
149    /// - Access permissions are violated (e.g., writing to read-only pages)
150    /// - Page table entries need to be populated or updated
151    ///
152    /// **Note**: This enum may be extended with additional fields in the future.
153    /// Use `..` in pattern matching to ensure forward compatibility.
154    NestedPageFault {
155        /// Guest physical address that caused the fault
156        addr: GuestPhysAddr,
157        /// Type of access that was attempted (read/write/execute)
158        access_flags: MappingFlags,
159    },
160
161    /// The guest VCpu has executed a halt instruction and is now idle.
162    ///
163    /// This typically occurs when the guest OS has no work to do and is
164    /// waiting for interrupts or other events to wake it up.
165    Halt,
166
167    /// Request to bring up a secondary CPU core.
168    ///
169    /// This exit reason is used during the multi-core VM boot process when
170    /// the primary CPU requests that a secondary CPU be started. The specific
171    /// mechanism varies by architecture:
172    ///
173    /// - **ARM**: PSCI (Power State Coordination Interface) calls
174    /// - **x86**: SIPI (Startup Inter-Processor Interrupt)
175    /// - **RISC-V**: SBI (Supervisor Binary Interface) calls
176    CpuUp {
177        /// Target CPU identifier to be started
178        ///
179        /// Format varies by architecture:
180        /// - **AArch64**: MPIDR register affinity fields  
181        /// - **x86_64**: APIC ID of the target CPU
182        /// - **RISC-V**: Hart ID of the target CPU
183        target_cpu: u64,
184        /// Guest physical address where the secondary CPU should begin execution
185        entry_point: GuestPhysAddr,
186        /// Argument to pass to the secondary CPU
187        ///
188        /// - **AArch64**: Value to set in `x0` register at startup
189        /// - **RISC-V**: Value to set in `a1` register (`a0` gets the hartid)
190        /// - **x86_64**: Currently unused
191        arg: u64,
192    },
193
194    /// The guest VCpu has been powered down.
195    ///
196    /// This indicates the VCpu has executed a power-down instruction or
197    /// hypercall and should be suspended. The VCpu may be resumed later.
198    CpuDown {
199        /// Power state information (currently unused)
200        ///
201        /// Reserved for future use with PSCI_POWER_STATE or similar mechanisms
202        _state: u64,
203    },
204
205    /// The guest has requested system-wide shutdown.
206    ///
207    /// This indicates the entire virtual machine should be powered off,
208    /// not just the current VCpu.
209    SystemDown,
210
211    /// No special handling required - the VCpu handled the exit internally.
212    ///
213    /// This provides an opportunity for the hypervisor to:
214    /// - Check virtual device states
215    /// - Process pending interrupts  
216    /// - Handle background tasks
217    /// - Perform scheduling decisions
218    ///
219    /// The VCpu can typically be resumed immediately after these checks.
220    Nothing,
221
222    /// VM entry failed due to invalid VCpu state or configuration.
223    ///
224    /// This corresponds to `KVM_EXIT_FAIL_ENTRY` in KVM and indicates that
225    /// the hardware virtualization layer could not successfully enter the guest.
226    ///
227    /// The failure reason contains architecture-specific diagnostic information.
228    FailEntry {
229        /// Hardware-specific failure reason code
230        ///
231        /// Interpretation depends on the underlying virtualization technology
232        /// and CPU architecture. Consult architecture documentation for details.
233        hardware_entry_failure_reason: u64,
234    },
235
236    /// The guest is attempting to send an Inter-Processor Interrupt (IPI).
237    ///
238    /// IPIs are used for inter-CPU communication in multi-core systems.
239    /// This does **not** include Startup IPIs (SIPI), which are handled
240    /// by the [`AxVCpuExitReason::CpuUp`] variant.
241    SendIPI {
242        /// Target CPU identifier to receive the IPI
243        ///
244        /// This field is invalid if `send_to_all` or `send_to_self` is true.
245        target_cpu: u64,
246        /// Auxiliary field for complex target CPU specifications
247        ///
248        /// Currently used only on AArch64 where:
249        /// - `target_cpu` contains `Aff3.Aff2.Aff1.0`
250        /// - `target_cpu_aux` contains a bitmask for `Aff0` values
251        target_cpu_aux: u64,
252        /// Whether to broadcast the IPI to all CPUs except the sender
253        send_to_all: bool,
254        /// Whether to send the IPI to the current CPU (self-IPI)
255        send_to_self: bool,
256        /// IPI vector/interrupt number to deliver
257        vector: u64,
258    },
259}
