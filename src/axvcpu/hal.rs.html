1/// Hardware abstraction layer interfaces for VCpu operations.
2///
3/// This trait defines the interfaces that the underlying software (kernel or hypervisor)
4/// must implement to support VCpu operations such as interrupt handling and memory management.
5pub trait AxVCpuHal {
6    /// Memory management interfaces required by the VCpu subsystem.
7    /// Must implement the AxMmHal trait from the axaddrspace crate.
8    type MmHal: axaddrspace::AxMmHal;
9
10    /// Fetches the current interrupt (IRQ) number from the hardware.
11    fn irq_fetch() -> usize {
12        0
13    }
14
15    /// Dispatches an interrupt request (IRQ) to the underlying host OS.
16    ///
17    /// This function should handle the actual interrupt processing and delegation
18    /// to the appropriate interrupt handler in the host system.
19    ///
20    /// # Implementation Required
21    ///
22    /// The default implementation panics as this function **must** be implemented
23    /// by the underlying hypervisor or kernel to provide proper interrupt handling.
24    ///
25    /// # Safety
26    ///
27    /// Implementations should ensure proper interrupt handling and avoid
28    /// reentrancy issues during interrupt processing.
29    fn irq_hanlder() {
30        unimplemented!("irq_handler is not implemented");
31    }
32}
