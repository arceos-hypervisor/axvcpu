9/// Immutable configuration data for a virtual CPU.
10///
11/// This structure contains the constant properties of a VCpu that don't change
12/// after creation, such as CPU affinity settings and identifiers.
13struct AxVCpuInnerConst {
14    /// Unique identifier of the VM this VCpu belongs to
15    vm_id: VMId,
16    /// Unique identifier of this VCpu within its VM
17    vcpu_id: VCpuId,
18    /// Physical CPU ID that has priority to run this VCpu
19    /// Used for CPU affinity optimization
20    favor_phys_cpu: usize,
21    /// Bitmask of physical CPUs that can run this VCpu
22    /// If `None`, the VCpu can run on any available physical CPU
23    /// Similar to Linux CPU_SET functionality
24    phys_cpu_set: Option<usize>,
25}

27/// Represents the current execution state of a virtual CPU.
28///
29/// The VCpu follows a strict state machine:
30/// Created → Free → Ready → Running
31///
32/// Invalid state is used when errors occur during state transitions.
33#[derive(Clone, Copy, Debug, PartialEq, Eq)]
34pub enum VCpuState {
35    /// Invalid state - indicates an error occurred during state transition
36    Invalid = 0,
37    /// Initial state after VCpu creation, not yet initialized
38    Created = 1,
39    /// VCpu is initialized and ready to be bound to a physical CPU
40    Free = 2,
41    /// VCpu is bound to a physical CPU and ready for execution
42    Ready = 3,
43    /// VCpu is currently executing on a physical CPU
44    Running = 4,
45    /// VCpu execution is blocked (waiting for I/O, etc.)
46    Blocked = 5,
47}

49/// Mutable runtime state of a virtual CPU.
50///
51/// This structure contains data that changes during VCpu execution,
52/// protected by RefCell for interior mutability.
53pub struct AxVCpuInnerMut {
54    /// Current execution state of the VCpu
55    state: VCpuState,
56}

58/// Architecture-independent virtual CPU implementation.
59///
60/// This is the main VCpu abstraction that provides a unified interface for
61/// managing virtual CPUs across different architectures. It delegates
62/// architecture-specific operations to implementations of the `AxArchVCpu` trait.
63///
64/// Note that:
65/// - This struct handles internal mutability itself, almost all the methods are `&self`.
66/// - This struct is not thread-safe. It's caller's responsibility to ensure the safety.
67pub struct AxVCpu<A: AxArchVCpu> {
68    /// Immutable VCpu configuration (VM ID, CPU affinity, etc.)
69    inner_const: AxVCpuInnerConst,
70    /// Mutable VCpu state protected by RefCell for safe interior mutability
71    inner_mut: RefCell<AxVCpuInnerMut>,
72    /// Architecture-specific VCpu implementation
73    ///
74    /// Uses UnsafeCell instead of RefCell because RefCell guards cannot be
75    /// dropped during VCpu execution (when control is transferred to guest)
76    arch_vcpu: UnsafeCell<A>,
77}

79impl<A: AxArchVCpu> AxVCpu<A> {
80    /// Creates a new virtual CPU instance.
81    ///
82    /// Initializes a VCpu with the given configuration and creates the underlying
83    /// architecture-specific implementation. The VCpu starts in the `Created` state.
84    ///
85    /// # Arguments
86    ///
87    /// * `vm_id` - Unique identifier of the VM this VCpu belongs to
88    /// * `vcpu_id` - Unique identifier for this VCpu within the VM
89    /// * `favor_phys_cpu` - Physical CPU ID that should preferentially run this VCpu
90    /// * `phys_cpu_set` - Optional bitmask of allowed physical CPUs (None = no restriction)
91    /// * `arch_config` - Architecture-specific configuration for VCpu creation
92    ///
93    /// # Returns
94    ///
95    /// Returns `Ok(AxVCpu)` on success, or an error if architecture-specific creation fails.
96    pub fn new(
97        vm_id: VMId,
98        vcpu_id: VCpuId,
99        favor_phys_cpu: usize,
100        phys_cpu_set: Option<usize>,
101        arch_config: A::CreateConfig,
102    ) -> AxResult<Self> {
103        Ok(Self {
104            inner_const: AxVCpuInnerConst {
105                vm_id,
106                vcpu_id,
107                favor_phys_cpu,
108                phys_cpu_set,
109            },
110            inner_mut: RefCell::new(AxVCpuInnerMut {
111                state: VCpuState::Created,
112            }),
113            arch_vcpu: UnsafeCell::new(A::new(vm_id, vcpu_id, arch_config)?),
114        })
115    }
116
117    /// Sets up the VCpu for execution.
118    ///
119    /// Configures the VCpu's entry point, memory management (EPT root), and any
120    /// architecture-specific setup. Transitions the VCpu from `Created` to `Free` state.
121    pub fn setup(
122        &self,
123        entry: GuestPhysAddr,
124        ept_root: HostPhysAddr,
125        arch_config: A::SetupConfig,
126    ) -> AxResult {
127        self.manipulate_arch_vcpu(VCpuState::Created, VCpuState::Free, |arch_vcpu| {
128            arch_vcpu.set_entry(entry)?;
129            arch_vcpu.set_ept_root(ept_root)?;
130            arch_vcpu.setup(arch_config)?;
131            Ok(())
132        })
133    }
134
135    /// Returns the unique identifier of this VCpu.
136    pub const fn id(&self) -> VCpuId {
137        self.inner_const.vcpu_id
138    }
139
140    /// Get the id of the VM this vcpu belongs to.
141    pub const fn vm_id(&self) -> VMId {
142        self.inner_const.vm_id
143    }
144
145    /// Returns the preferred physical CPU for this VCpu.
146    ///
147    /// This is used for CPU affinity optimization - the scheduler should
148    /// preferentially run this VCpu on the returned physical CPU ID.
149    ///
150    /// # Note
151    ///
152    /// Currently unused in the implementation but reserved for future
153    /// scheduler optimizations.
154    pub const fn favor_phys_cpu(&self) -> usize {
155        self.inner_const.favor_phys_cpu
156    }
157
158    /// Returns the set of physical CPUs that can run this VCpu.
159    pub const fn phys_cpu_set(&self) -> Option<usize> {
160        self.inner_const.phys_cpu_set
161    }
162
163    /// Checks if this VCpu is the Bootstrap Processor (BSP).
164    ///
165    /// By convention, the VCpu with ID 0 is always considered the BSP,
166    /// which is responsible for system initialization in multi-core VMs.
167    pub const fn is_bsp(&self) -> bool {
168        self.inner_const.vcpu_id == 0
169    }
170
171    /// Gets the current execution state of the VCpu.
172    pub fn state(&self) -> VCpuState {
173        self.inner_mut.borrow().state
174    }
175
176    /// Set the state of the VCpu.
177    /// # Safety
178    /// This method is unsafe because it may break the state transition model.
179    /// Use it with caution.
180    pub unsafe fn set_state(&self, state: VCpuState) {
181        self.inner_mut.borrow_mut().state = state;
182    }
183
184    /// Execute a block with the state of the VCpu transitioned from `from` to `to`. If the current state is not `from`, return an error.
185    ///
186    /// The state will be set to [`VCpuState::Invalid`] if an error occurs (including the case that the current state is not `from`).
187    ///
188    /// The state will be set to `to` if the block is executed successfully.
189    pub fn with_state_transition<F, T>(&self, from: VCpuState, to: VCpuState, f: F) -> AxResult<T>
190    where
191        F: FnOnce() -> AxResult<T>,
192    {
193        let mut inner_mut = self.inner_mut.borrow_mut();
194        if inner_mut.state != from {
195            inner_mut.state = VCpuState::Invalid;
196            ax_err!(
197                BadState,
198                format!("VCpu state is not {:?}, but {:?}", from, inner_mut.state)
199            )
200        } else {
201            let result = f();
202            inner_mut.state = if result.is_err() {
203                VCpuState::Invalid
204            } else {
205                to
206            };
207            result
208        }
209    }
210
211    /// Execute a block with the current VCpu set to `&self`.
212    pub fn with_current_cpu_set<F, T>(&self, f: F) -> T
213    where
214        F: FnOnce() -> T,
215    {
216        if get_current_vcpu::<A>().is_some() {
217            panic!("Nested vcpu operation is not allowed!");
218        } else {
219            unsafe {
220                set_current_vcpu(self);
221            }
222            let result = f();
223            unsafe {
224                clear_current_vcpu::<A>();
225            }
226            result
227        }
228    }
229
230    /// Execute an operation on the architecture-specific VCpu, with the state transitioned from `from` to `to` and the current VCpu set to `&self`.
231    ///
232    /// This method is a combination of [`AxVCpu::with_state_transition`] and [`AxVCpu::with_current_cpu_set`].
233    pub fn manipulate_arch_vcpu<F, T>(&self, from: VCpuState, to: VCpuState, f: F) -> AxResult<T>
234    where
235        F: FnOnce(&mut A) -> AxResult<T>,
236    {
237        self.with_state_transition(from, to, || {
238            self.with_current_cpu_set(|| f(self.get_arch_vcpu()))
239        })
240    }
241
242    /// Transition the state of the VCpu. If the current state is not `from`, return an error.
243    pub fn transition_state(&self, from: VCpuState, to: VCpuState) -> AxResult {
244        self.with_state_transition(from, to, || Ok(()))
245    }
246
247    /// Get the architecture-specific VCpu.
248    #[allow(clippy::mut_from_ref)]
249    pub fn get_arch_vcpu(&self) -> &mut A {
250        unsafe { &mut *self.arch_vcpu.get() }
251    }
252
253    /// Run the VCpu.
254    pub fn run(&self) -> AxResult<AxVCpuExitReason> {
255        self.transition_state(VCpuState::Ready, VCpuState::Running)?;
256        self.manipulate_arch_vcpu(VCpuState::Running, VCpuState::Ready, |arch_vcpu| {
257            arch_vcpu.run()
258        })
259    }
260
261    /// Bind the VCpu to the current physical CPU.
262    pub fn bind(&self) -> AxResult {
263        self.manipulate_arch_vcpu(VCpuState::Free, VCpuState::Ready, |arch_vcpu| {
264            arch_vcpu.bind()
265        })
266    }
267
268    /// Unbind the VCpu from the current physical CPU.
269    pub fn unbind(&self) -> AxResult {
270        self.manipulate_arch_vcpu(VCpuState::Ready, VCpuState::Free, |arch_vcpu| {
271            arch_vcpu.unbind()
272        })
273    }
274
275    /// Sets the entry address of the VCpu.
276    pub fn set_entry(&self, entry: GuestPhysAddr) -> AxResult {
277        self.get_arch_vcpu().set_entry(entry)
278    }
279
280    /// Sets the value of a general-purpose register according to the given index.
281    pub fn set_gpr(&self, reg: usize, val: usize) {
282        self.get_arch_vcpu().set_gpr(reg, val);
283    }
284
285    /// Inject an interrupt to the VCpu.
286    pub fn inject_interrupt(&self, vector: usize) -> AxResult {
287        self.get_arch_vcpu().inject_interrupt(vector)
288    }
289
290    /// Sets the return value of the VCpu.
291    pub fn set_return_value(&self, val: usize) {
292        self.get_arch_vcpu().set_return_value(val);
293    }
294}

296#[percpu::def_percpu]
297static mut CURRENT_VCPU: Option<*mut u8> = None;
298
299/// Get the current VCpu on the current physical CPU.
300///
301/// It's guaranteed that each time before a method of [`AxArchVCpu`] is called, the current VCpu is set to the corresponding [`AxVCpu`].
302/// So methods of [`AxArchVCpu`] can always get the [`AxVCpu`] containing itself by calling this method.
303pub fn get_current_vcpu<'a, A: AxArchVCpu>() -> Option<&'a AxVCpu<A>> {
304    unsafe {
305        CURRENT_VCPU
306            .current_ref_raw()
307            .as_ref()
308            .copied()
309            .and_then(|p| (p as *const AxVCpu<A>).as_ref())
310    }
311}
312
313/// Get a mutable reference to the current VCpu on the current physical CPU.
314///
315/// See [`get_current_vcpu`] for more details.
316pub fn get_current_vcpu_mut<'a, A: AxArchVCpu>() -> Option<&'a mut AxVCpu<A>> {
317    unsafe {
318        CURRENT_VCPU
319            .current_ref_mut_raw()
320            .as_mut()
321            .copied()
322            .and_then(|p| (p as *mut AxVCpu<A>).as_mut())
323    }
324}

326/// Set the current VCpu on the current physical CPU.
327///
328/// # Safety
329/// This method is marked as unsafe because it may result in unexpected behavior if not used properly.
330/// Do not call this method unless you know what you are doing.
331pub unsafe fn set_current_vcpu<A: AxArchVCpu>(vcpu: &AxVCpu<A>) {
332    unsafe {
333        CURRENT_VCPU
334            .current_ref_mut_raw()
335            .replace(vcpu as *const _ as *mut u8);
336    }
337}

339/// Clear the current vcpu on the current physical CPU.
340///
341/// # Safety
342/// This method is marked as unsafe because it may result in unexpected behavior if not used properly.
343/// Do not call this method unless you know what you are doing.    
344pub unsafe fn clear_current_vcpu<A: AxArchVCpu>() {
345    unsafe {
346        CURRENT_VCPU.current_ref_mut_raw().take();
347    }
